⚙️ IMPORTANT STEPS | خطوات مهمة

═══════════════════════════════════════════════════════════════════════════════

STEP 1: احصل على Google Gemini API Key
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  1. اذهب إلى: https://aistudio.google.com/app/apikey
  2. اضغط "Create API Key" 
  3. انسخ المفتاح

STEP 2: أضف المفتاح إلى .env.local
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  افتح ملف .env.local وأضف:
  VITE_GEMINI_API_KEY=your_api_key_here

STEP 3: شغّل الموقع
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  npm run dev

🐛 TROUBLESHOOTING | استكشاف الأخطاء

═══════════════════════════════════════════════════════════════════════════════

المساعد الذكي لا يعمل؟
  → تحقق من وجود VITE_GEMINI_API_KEY في .env.local
  → تأكد من صحة المفتاح
  → امسح الكاش وأعد تحميل الصفحة (Ctrl+Shift+Delete)

الموقع يحمل ببطء؟
  → امسح الكاش المتصفح
  → استخدم متصفح حديث
  → جرب في وضع التصفح الخاص

الأحرف العربية غير صحيحة؟
  → تأكد من ترميز UTF-8 في المتصفح
  → امسح الكاش القديم
